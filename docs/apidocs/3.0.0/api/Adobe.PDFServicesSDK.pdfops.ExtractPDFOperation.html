Class ExtractPDFOperation

An Operation that extracts pdf elements such as text, images, tables in a structured format from a PDF.

Examples

Sample Usage:

string credentialsFilePath = Directory.GetCurrentDirectory()
    + Path.DirectorySeparatorChar + "pdfservices-api-credentials.json";
Credentials credentials = Credentials.ServiceAccountCredentialsBuilder()
    .FromFile(credentialsFilePath).Build();
ExecutionContext context = ExecutionContext.Create(credentials);
ExtractPDFOperation extractPdfOperation = ExtractPDFOperation.CreateNew();
FileRef sourceFileRef = FileRef.CreateFromLocalFile(@"extractPDFInput.pdf");
extractPdfOperation.SetInputFile(sourceFileRef);
ExtractPDFOptions extractPdfOptions = ExtractPDFOptions.ExtractPDFOptionsBuilder()
    .AddElementsToExtract(new List <ExtractElementType>(new []{ ExtractElementType.TEXT}))
    .Build();
extractPdfOperation.SetOptions(extractPdfOptions);
FileRef result = extractPdfOperation.Execute(context);
result.SaveAs("ExtractTextInfoFromPDF.zip");
